Dhalsim


To use this module, you need to js files from the "dist/" folder


Breakpoints

    dhalsim.init();
    dhalsim.isMobile();
    // use breakpoint < 768px

    dhalsim.isTablet();
    // use breakpoint >= 768px

    dhalsim.isDesktop();
    // use breakpoint >= 960px

    dhalsim.isWide();
    // use breakpoint >= 1200px

    dhalsim.beyondTablet();
    // use >= 900px || >= 1200px

    dhalsim.beyondMobile();
    // use >= 768px || (>= 900px || >= 1200px)

Global

    var dhalsim = new window.dhalsim.default();
    dhalsim.init();

    dhalsim.isMobile();

Init (ES6)

  import dhalsim from 'dhalsim-js';

  const dhalsim = new Dhalsim();
  dhalsim.init();

  dhalsim.isMobile();

Init (CommonJs)

    var Dhalsim = require('dhalsim-js');

    const dhalsim = new Dhalsim();
    dhalsim.init();

    dhalsim.isMobile();

Init (AMD)

    define(['dhalsim-js'], function(Dhalsim){
      var dhalsim = new Dhalsim();
      dhalsim.init();

      dhalsim.isMobile();
    });
