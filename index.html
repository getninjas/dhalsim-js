MediaQuery


To use this module, you need to import both css and js files from the "dist/" folder


Breakpoints

    mediaQuery.isMobile();
    // use breakpoint < 768px

    mediaQuery.isTablet();
    // use breakpoint >= 768px

    mediaQuery.isDesktop();
    // use breakpoint >= 960px

    mediaQuery.isWide();
    // use breakpoint >= 1200px

    mediaQuery.beyondTablet();
    // use >= 900px || >= 1200px

    mediaQuery.beyondMobile();
    // use >= 768px || (>= 900px || >= 1200px)

Global

    var mediaQuery = new window.mediaQuery.default();

    mediaQuery.isMobile();

Init (ES6)

  import MediaQuery from 'media-query';

  const mediaQuery = new MediaQuery();

  mediaQuery.isMobile();

Init (CommonJs)

    var MediaQuery = require('media-query');

    const mediaQuery = new MediaQuery();

    mediaQuery.isMobile();

Init (AMD)

    define(['media-query'], function(MediaQuery){
      var mediaQuery = new MediaQuery();

      mediaQuery.isMobile();
    });
